\section{Holomorphic Functions}

\begin{definition}
    Let $f:U \subset \mathbb C \to \mathbb C$. We call $f$ holomorphic
    in $a \in U$ if the following limit exists:
    \begin{displaymath}
        \lim_{h\to 0} \frac{f(a+h) - f(a)}{h}, \quad (h \in \mathbb C).
    \end{displaymath}
    
\end{definition}
